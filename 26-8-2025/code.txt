CREATE TABLE Worker (
	WORKER_ID INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
	FIRST_NAME CHAR(25),
	LAST_NAME CHAR(25),
	SALARY INT(15),
	JOINING_DATE DATETIME,
	DEPARTMENT CHAR(25)
);

SELECT SUM(SALARY) AS total_salary FROM Worker;

SELECT AVG(SALARY) AS average_salary FROM Worker;

SELECT 
  WORKER_ID,
  FIRST_NAME,
  LAST_NAME,
  SALARY,
  CASE 
    WHEN SALARY >= 300000 THEN 'High'
    WHEN SALARY BETWEEN 100000 AND 299999 THEN 'Medium'
    ELSE 'Low'
  END AS GRADE
FROM Worker;

SELECT 
  FIRST_NAME,
  DEPARTMENT,
  CASE 
    WHEN DEPARTMENT IN ('HR', 'Admin') THEN 'Non-Technical'
    ELSE 'Technical'
  END AS Dept_Type
FROM Worker;
